% 1.12
\begin{exercise}
    \begin{enumerate}
        \item \(\ideala \subseteq (\ideala : \idealb)\)
        \item \((\ideala : \idealb) \idealb \subseteq \ideala\)
        \item \(((\ideala : \idealb) : \idealc) = (\ideala : \idealb \idealc) = ((\ideala : \idealc) : \idealb)\)
        \item \((\bigcap_i \ideala_i : \idealb) = \bigcap_i (\ideala_i : \idealb)\)
        \item \((\ideala : \sum_i \idealb_i) = \bigcap_i (\ideala : \idealb_i)\).
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate}
    \item
        For \(a \in \ideala\), \(a \idealb \subseteq \ideala\) since \(\ideala\) is an ideal.
    \item
        Since \(xb \in (\ideala : \idealb) \idealb\) iff \(x \idealb \subseteq \ideala\) and \(b \in \idealb\), \(xb \in \ideala\).
    \item
        Since \(\idealb \idealc = \idealc \idealb\) we only need to show the first equality.
        Observe that \(x \in ((\ideala : \idealb) : \idealc)\) iff \(x \idealc \subseteq (\ideala : \idealb)\), that is for all \(xc \in x \idealc\) with \(c \in \idealc\), \(xc \idealb \subseteq \ideala\).
        This is true iff \(x \idealc \idealb \subseteq \ideala\).
    \item
        \(x \idealb \subseteq \bigcap_i \ideala_i\) iff \(x \idealb \subseteq \ideala_i\) for all \(i\).
    \item
        Note \(x \sum_i \idealb_i = \sum_i x \idealb_i\).
        Then \(\sum_i x \idealb_i \subseteq \ideala\) iff \(x \idealb_i \subseteq \ideala\) for all \(i\).
\end{enumerate}

% 1.13
\begin{exercise}
    \begin{enumerate}
        \item \label{ex-1.13.1} \(\rad{\ideala} \supseteq \ideala\)
        \item \(\rad{\rad{\ideala}} = \rad{\ideala}\)
        \item \(\rad{\mathfrak{ab}} = \rad{\ideala \cap \idealb} = \rad{\ideala} \cap \rad{\idealb}\)
        \item \(\rad{\ideala} = (1) \iff \ideala = (1)\)
        \item \(\rad{\ideala + \idealb} = \rad{\rad{\ideala} + \rad{\idealb}}\)
        \item if \(\idealp\) is prime, \(\rad{\idealp^n} = \idealp\) for all \(n > 0\).
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate}
    \item
        If \(x \in \ideala\), then \(x^1 \in \ideala\) so \(x \in \rad{\ideala}\).
    \item
        \(x \in \rad{\rad{\ideala}}\) iff \((x^n)^m \in \ideala\) iff \(x^{nm} \in \ideala\) iff \(x \in \rad{\ideala}\).
    \item
        If \(x^n \in \ideala \idealb\) then \(x^n \in \ideala \cap \idealb\).
        If \(x^n \in \ideala \cap \idealb\) then \(x^n \in \ideala\) and \(x^n \in \idealb\).
        If \(x^n \in \ideala\) and \(x^m \in \idealb\) then \(x^{mn} \in \ideala \idealb\).
    \item
        If \(1^n \in \ideala\) then \(1 \in \ideala\).
        The converse comes from \ref{ex-1.13.1}.
    \item
        By 1., \(\rad{\ideala + \idealb} \subseteq \rad{\rad{\ideala} + \rad{\idealb}}\).
        If \(x^n = y + z\) where \(y^k \in \ideala\) and \(z^m \in \idealb\) then \(x^{n (k + m)} = (y + z)^{k + m}\) has every term in its binomial expansion with either its \(y\) term at least \(k\) or its \(z\) term at least \(m\), and so \(x^{n (k+m)} \in \ideala + \idealb\).
    \item
        If \(x^m \in \idealp^n\) then \(x^m \in \idealp\) so \(x \in \idealp\).
        Conversely if \(x \in \idealp\) then \(x^n \in \idealp^n\) so \(x \in \rad{\idealp^n}\).
\end{enumerate}

\begin{exercise}
    If \(\ideala_1, \ideala_2\) are ideals of \(A\) and if \(\idealb_1, \idealb_2\) are ideals of \(B\), then
    \[
        \begin{array}{c c}
            (\ideala_1 + \ideala_2)^e = \ideala_1^e + \ideala_2^e, &
            (\idealb_1 + \idealb_2)^c \supseteq \idealb_1^c + \idealb_2^c, \\
            (\ideala_1 \cap \ideala_2)^e \subseteq \ideala_1^e \cap \ideala_2^e, &
            (\idealb_1 \cap \idealb_2)^c = \idealb_1^c \cap \idealb_2^c, \\
            (\ideala_1 \ideala_2)^e = \ideala_1^e \ideala_2^e, &
            (\idealb_1 \idealb_2)^c \supseteq \idealb_1^c \idealb_2^c, \\
            (\ideala_1 : \ideala_2)^e \subseteq (\ideala_1^e : \ideala_2^e), &
            (\idealb_1 : \idealb_2)^c \subseteq (\idealb_1^c : \idealb_2^c), \\
            \rad{\ideala}^e \subseteq \rad{\ideala^e} &
            (\rad{\idealb})^c = \rad{\idealb^c}.
        \end{array}
    \]
\end{exercise}

\noindent\textbf{Solution:}
Note that extension is monotone.
Hence \(\ideala_1^e \subseteq (\ideala_1 + \ideala_2)^e\) and \(\ideala_2^e \subseteq (\ideala_1 + \ideala_2)^e\).
Hence \(\ideala_1^e + \ideala_2^e \subseteq (\ideala_1 + \ideala_2)^e\).
Conversely, let \(\sum_i f(a_i + b_i) \in (\ideala_1 + \ideala_2)^e\).
Then \(\sum_i f(a_i + b_i) = (\sum_i f(a_i)) + (\sum_i f(b_i)) \in \ideala_1^e + \ideala_2^e\).
Next, notice that \((\ideala_1 \cap \ideala_2)^e \subseteq \ideala_1^e\) and \((\ideala_1 \cap \ideala_2)^e \subseteq \ideala_2^e\).
Hence \((\ideala_1 \cap \ideala2)^e \subseteq \ideala_1^e \cap \ideala_2^e\).
Next, let \(\sum_i f(a_i b_i) \in (\ideala_1 \ideala_2)^e\).
Then \(\sum_i f(a_i b_i) = \sum_i f(a_i) f(b_i) \in \ideala_1^e \ideala_2^e\).
Conversely let \(\sum_i (\sum_j f(a_{ij})) (\sum_k f(b_{ik})) \in \ideala_1^e \ideala_2^e\).
Then \(\sum_i (\sum_j f(a_{ij})) (\sum_k f(b_{ik})) = \sum_i \sum_j \sum_k f(a_{ij}) f(b_{ik}) = \sum_i \sum_j \sum_k f(a_{ij}b_{ik}) \in (\ideala_1 \ideala_2)^e\).
Next, let \(\sum_i f(a_i) \in (\ideala_1 : \ideala_2)^e\) such that \(a_i \ideala_2 \subseteq \ideala_1\).
Then \((f(a_i))\ideala_2^e = (a_i)^e\ideala_2^e = (a_i \ideala_2)^e \subseteq a_1^e\) so \(f(a_i) \in (\ideala_1^e : \ideala_2^e)\).
Next let \(\sum_i f(a_i) \in \rad{\idealb}^e\)
Then for some \(n_i\), \(f(a_i)^{n_i} \in \ideala^e\) so \(\sum_i f(a_i) \in \rad{\ideala^e}\).

Let \(a + b \in \idealb_1^c + \idealb_2^c\).
Then \(f(a) \in \idealb_1, f(b) \in \idealb_2\) so \(f(a + b) \in \idealb_1 + \idealb_2\) hence \(a + b \in (\idealb_1 + \idealb_2)^c\).
Next let \(b \in (\idealb_1 \cap \idealb_2)^c\).
Then \(f(b) \in \idealb_1 \cap \idealb_2\).
Hence \(b \in \idealb_1^c \cap \idealb_2^c\).
Conversely if \(b \in \idealb_1^c \cap \idealb_2^c\) then \(f(b) \in \idealb_1\) and \(f(b) \in \idealb_2\) so \(b \in (\idealb_1 \cap \idealb_2)^c\).
Next let \(\sum_i a_i b_i \in \idealb_1^c \idealb_2^c\).
Then \(f(a_i) \in \idealb_1\) and \(f(b_i) \in \idealb_2\) so \(f(\sum_i a_i b_i) \in \idealb_1 \idealb_2\).
Hence \(\sum_i a_i b_i \in (\idealb_1 \idealb_2)^c\).
Next let \(b \in (\idealb_1 : \idealb_2)^c\).
Then \(f(b) \idealb_2 \subseteq \idealb_1\), that is for each \(c \in \idealb_2\), \(f(b) c \in \idealb_1\).
If \(c' \in \idealb_2^c\) so that \(f(c') = c\) then \(f(b c') \in \idealb_1\), so \(bc' \in \idealb_1^c\).
Hence \(b \in (\idealb_1^c : \idealb_2^c)\).
Next, if \(b \in (\rad{\idealb})^c\) so that \(f(b)^n \in \idealb\), then \(f(b^n) \in \idealb\) so \(b^n \in \idealb^c\), so \(b \in \rad{\idealb^c}\).
Conversely, if \(b \in \rad{\idealb^c}\) so that \(b^n \in \idealb^c\), then \(f(b)^n \in \idealb\) so \(f(b) \in \rad{\idealb}\) so \(b \in (\rad{\idealb})^c\).

\begin{exercise} \label{ex1-1}
    Let \(x\) be a nilpotent element of a ring \(A\).
    Show that \(1 + x\) is a unit in \(A\).
    Deduce that the sum of a nilpotent element and a unit is a unit.
\end{exercise}

\noindent\textbf{Solution:}
Suppose \(x^n = 0\).
Notice that
\[
    (1 + x)(\prod_{i=0}^n (-1)^i x^i) = 1 + x - x - x^2 + x^2 + x^3 \cdots \pm x^n = 1 \pm x^n = 1.
\]
If \(u\) is a unit then \(u^{-1} x\) is nilpotent, so \(1 + u^{-1} x\) is a unit, hence \(u + x\) is.

\begin{exercise}
    Let \(A\) be a ring and let \(A[x]\) be the ring of polynomials in an indeterminate \(x\), with coefficients in \(A\).
    Let \(f = a_0 + a_1 x + \cdots + a_n x^n\). Prove that
    \begin{enumerate}
        \item \label{ex1-2-1}
            \(f\) is a unit \(\iff\) \(a_0\) is a unit in \(A\) and \(a_1, \cdots, a_n\) are nilpotent.
        \item
            \(f\) is nilpotent \(\iff\) \(a_0, a_1, \cdots, a_n\) are nilpotent.
        \item
            \(f\) is a zero-divisor \(\iff\) there exists \(a \neq 0\) in \(A\) such that \(a f = 0\).
        \item
            \(f\) is said to be primitive if \((a_0, a_1, \cdots, a_n) = (1)\).
            Prove that if \(f, g \in A[x]\), then \(fg\) is primitive \(\iff\) f and g are primitive.
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate} \label{ex1-2}
    \item
        If \(a_0\) is a unit and \(a_i\) are nilpotent for \(i \geq 1\), then \(a_i x^i\) are nilpotent and their sum is nilpotent, hence by \ref{ex1-1} \(f\) is a unit.

        Conversely suppose \(f\) is a unit with inverse \(g\) (coefficients indexed with \(b_i\)).
        Then \(a_0 b_0 = 1\).
        Next let \(\idealp\) be a prime ideal.
        The map \(\varphi : A[x] \to (A/\idealp)[x]\) has \(\varphi(f)\varphi(g) = 1\).
        Hence the degree of \(\varphi(f)\) and \(\varphi(g)\) are both \(0\) since \((A/\idealp)[x]\) is an integral domain.
        Thus \(\phi(a_i) = 0\) for all \(i \geq 1\), that is \(a_i \in \idealp\).
        Since this is for any prime ideal, the \(a_i\) are in all prime ideals, and hence are nilpotent.
    \item
        Similarly to above, if the \(a_i\) are nilpotent, their product with indeterminates are, and their sum is.

        Let \(f\) be nilpotent with \(f^n = 0\).
        Then \(a_0^n = 0\) so \(a_0\) is nilpotent.
        By exercise \ref{ex1-1}, \(f + 1\) is a unit, and by part \ref{ex1-2-1} the \(a_i\) are nilpotent for \(i \geq 1\).
    \item
        Let \(g \neq 0\) annihilate \(f\) with minimum degree \(m\).
        Then \(b_m a_n = 0\).
        Hence \(a_n g\) has degree less than m, and moreover \(a_n g\) annihilates \(f\).
        Hence \(a_n g = 0\).
        Now suppose for induction that \(a_{n-i} g = 0\) holds for all \(i \leq r\).
        Then \(a_{n-r-1} b_i = - \sum_{j > i} a_{n-j} b_j = 0\).
        Hence for all \(i\), \(a_i b_n = 0\).
        Thus \(b_n f = 0\).
    \item
        Suppose \(fg\) is primitive, and call its coefficients \(c_i\).
        It is sufficient to show \(f\) is primitive.
        Notice that \((a + b) \subseteq (a) + (b)\).
        Hence \((c_i) \subseteq \sum_{j \leq i} (a_j b_{i-j}) \subseteq \sum_{j \leq i} (a_j)\) for all \(i\).
        Thus \((1) = \sum_i (c_i) \subseteq \sum_i (a_j)\), and so \(f\) is primitive.

        Conversely suppose \(f\) and \(g\) are primitive.
        We will show that \(\idealc = (c_0, c_1, \cdots, c_k)\) is not contained in any prime ideal of \(A\).
        Suppose it were contained \(\idealp\).
        Then under the map \(\varphi : A[x] \to (A/\idealp)[x]\), \(\varphi(c_i) = 0\) and so \(\varphi(fg) = 0\).
        Since \((A/\idealp)[x]\) is a domain, either \(\varphi(f) = 0\) or \(\varphi(g) = 0\).
        However, since \((a_0, a_1, \cdots, a_n) \not\supseteq \idealp\) (since \(\idealp\) is proper), some \(a_i\) is not sent to zero.
        Similarly, some \(b_i\) is not sent to zero.
        This is a contradiction, and so \(\idealc\) is not contained in any prime ideal.
        Since every proper ideal is contained in a maximal (and hence prime) ideal, \(\idealc\) is not proper.
\end{enumerate}

\begin{exercise} \label{ex1-3}
    Generalise the results in Exercise \ref{ex1-2} to a polynomial ring \(A[x_1, \cdots, x_n]\) in several indeterminates.
\end{exercise}

\noindent\textbf{Solution:}
The proofs of part 1, 2, and 4 do not depend on being a single variable, so their proofs can be adapted exactly.
We will only show part 3.
Let \(f\) be a zero-divisor with annihilator \(g\).
Let \(d\) be greater than the maximum exponent of indeterminates in \(f\).
Consider the map \(\varphi : A[x_1, \cdots, x_n] \to A[y]\) sending \(x_i\) to \(y^{d^i}\).
Notice that \(\varphi\) restricted to the set of polynomials with maximum exponent indeterminates less than \(d\) is injective.
To see this, write the exponent of the image of each monomial in base \(d\), each monomial is mapped distinctly.
Then \(\varphi(fg) = 0\) so \(\varphi(f)\) is a zero-divisor.
Hence there is an \(a \neq 0\) such that \(\varphi(af) = 0\).
By injectivity of the restriction, \(af = 0\).

\begin{exercise}
    In the ring \(A[x]\), the Jacobson radical is equal to the nilradical.
\end{exercise}

\noindent\textbf{Solution:}
Since the set of maximal ideals is a subset of the set of prime ideals, the nilradical is contained in the Jacobson radical.
Conversely let \(f\) be in the Jacobson radical.
Then \(xf - 1\) is a unit, so by Exercise \ref{ex1-2} all coefficients of \(f\) are nilpotent.
Hence \(f\) is nilpotent.


\begin{exercise}
    Let \(A\) be a ring and let \(A[[x]]\) be the ring of formal power series \(f = \sum_{n=0}^\infty a_n x^n\) with coefficients in \(A\).
    Show that
    \begin{enumerate}
        \item \label{ex1-5-1}
            \(f\) is a unit in \(A[[x]]\) \(\iff\) \(a_0\) is a unit in \(A\).
        \item
            If \(f\) is nilpotent, then \(a_n\) is nilpotent for all \(n \geq 0\).
            Is the converse true?
        \item
            \(f\) belongs to the Jacobson radical of \(A[[x]]\) \(\iff\) \(a_0\) belongs to the Jacobson radical of \(A\).
        \item
            The contraction of a maximal ideal \(\idealm\) of \(A[[x]]\) is a maximal ideal of \(A\), and \(\idealm\) is generated by \(\idealm^c\) and \(x\).
        \item
            Every prime ideal of \(A\) is the contraction of a prime ideal of \(A[[x]]\).
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate}
    \item
        If \(f\) is a unit with inverse \(g = \sum_{n=0}^\infty b_n x^n\) then \(1 = a_0 b_0\) so \(a_0\) is a unit.

        Conversely, suppose \(a_0\) is a unit.
        We define the inverse \(g\) of \(f\) by induction.
        Let \(b_0\) be the inverse of \(a_0\).
        Next, assume we have the first \(n\) terms of \(g\) constructed so that the first \(n\) terms in their product is zero.
        Then \(c_n+1 = \sum_{i=0}^{n+1} a_i b_{n+1-i}\).
        We want \(c_n+1 = 0\) so we get that \(b_{n+1} a_0 = -\sum_{i=1}^{n+1} a_i b_{n+1-i}\).
        Since \(a_0\) is a unit, we can invert it to get \(-b_0 \sum_{i=1}^{n+1} a_i b_{n+1-i}\).
        This completes the construction of the inverse of \(f\).
    \item
        Let \(f\) be such that \(f^n = 0\).
        Then \(a_0^n = 0\) and so \(a_0\) is nilpotent.
        Now suppose for induction that \(a_i\) are nilpotent for all \(i \leq m\) with exponents \(n_i\).
        Since the nilradical is an ideal, we can define \(g = f - \sum_{i \leq m} a_i x^i\) and \(g\) will be nilpotent, say \(g^k = 0\).
        Then \(a_{m+1}^k = 0\), and so \(a_{m+1}\) is nilpotent.

        The converse does not hold, as we could have a power series where each coefficient is nilpotent, but the exponent for it to go to zero is unbounded.
        TODO make an actual example.
    \item
        \(f\) is in the Jacobson radical of \(A[[x]]\) iff \(1 - fg\) is a unit for all \(g \in A[[X]]\).
        This is true iff \(1 - a_0 b_0\) is a unit for all \(b_0 \in A\) by part \ref{ex1-5-1}.
        And finally this is true iff \(a_0\) is in the Jacobson radical of \(A\).
    \item
        Since \(x\) is in the Jacobson radical, it is it \(\idealm\).
        Then since \(A[[x]] / (x) = A\), we see that \(\idealm / (x) = \idealm^c\).
        Hence \(\idealm\) is generated by \(\idealm^c\) and \(x\).
        Moreover \(A[[x]]/\idealm = (A[[x]]/(x))/(\idealm/(x)) = A/\idealm^c\) and so \(\idealm^c\) is maximal.
    \item
        Let \(\idealp\) be a prime ideal of \(A\).
        Then \(\idealp^e + (x)\) is an ideal of \(A[[x]]\) and moreover \(A[[x]]/(\idealp^e + (x)) = A/\idealp\) so \(\idealp^e + (x)\) is prime.
        Moreover \(\idealp\) is the contraction of \(\idealp^e + (x)\).
\end{enumerate}

\begin{exercise}
    A ring \(A\) is such that every ideal not contained in the nilradical contains a nonzero idempotent (that is, an element such that \(e^2 = e \neq 0\)).
    Prove that the nilradical and the Jacobson radical of \(A\) are equal.
\end{exercise}

\noindent\textbf{Solution:}
The nilradical is contained in the Jacobson radical as every maximal ideal is prime, so we just need to show that the opposite inclusion holds.
Suppose \(a\) is in the Jacobson radical but not the nilradical.
Then \((a)\) is an ideal not contained in the nilradical, and so has a nonzero idempotent \(e = ab\) (for some \(b \in A\)).
Note also that since \(a\) is in the Jacobson radical, \(1 - e\) is a unit with inverse \(c\), so \(c - ce = 1\).
Then \(1 - ce = c\), so \(1 = (1 - e)(1 - ce) = 1 - e - ce + ce^2 = 1 - e\).
This contradicts \(e\) being nonzero!
So the Jacobson radical and nilradical are equal.

\begin{exercise} \label{ex1-7}
    Let \(A\) be a ring in which every element \(x\) satisfies \(x^n = x\) for some \(n > 1\) (depending on x).
    Show that every prime ideal is maximal.
\end{exercise}

\noindent\textbf{Solution:}
Let \(\idealp\) be prime.
Let \(x + \idealp\) be a nonzero element.
Then \(x (x^{n-1} - 1) + \idealp = 0 + \idealp\), so \(x^{n-1} \equiv 1 \pmod{\idealp}\).
Hence \(x + \idealp\) is a unit, so \(\idealp\) is maximal.

\begin{exercise} \label{ex1-8}
    Let \(A\) be a ring \(\neq 0\).
    Show that the set of prime ideals of \(A\) has minimal elements with respect to inclusion.
\end{exercise}

\noindent\textbf{Solution:}
We use Zorn's lemma.
Firstly, we know that the set of prime ideals is non-empty since it has a maximal ideal.
Let \(C\) be a chain in the poset of prime ideals.
The intersection \(\idealp = \cap C\) of this chain is an ideal.
Suppose \(a, b \notin \idealp\).
Then there is some \(\idealp_c \in C\) such that \(a, b \notin \idealp_c\).
Hence, \(ab \notin \idealp_c\), so \(ab \notin \idealp\).
Thus \(\idealp\) is prime.
Every chain has a lower bound, so Zorn's lemma tells us that there is a minimal element.

\begin{exercise}
    Let \(\ideala\) be an ideal \(\neq (1)\) in a ring \(A\).
    Show that \(\ideala = \rad{\ideala}\) \(\iff\) \(\ideala\) is an intersection of prime ideals.
\end{exercise}

\noindent\textbf{Solution:}
If \(\ideala = \rad{\ideala}\) then \(\ideala\) is the intersection of all ideals containing it.
Conversely, suppose \(\ideala\) is an intersection of prime ideals \(\idealp_i\).
Let \(x \in \rad{\ideala}\) so that \(x^n \in \ideala\).
Then for each \(\idealp_i\), \(x^n \in \idealp_i\) so \(x \in \idealp_i\).
Hence \(x \in \ideala\).

\begin{exercise}
    Let \(A\) be a ring, \(\nilrad\) its nilradical.
    Show that the following are equivalent:
    \begin{enumerate}
        \item
            \(A\) has exactly one prime ideal;
        \item
            every element of \(A\) is either a unit or nilpotent;
        \item
            \(A/\nilrad\) is a field.
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
If \(A\) has exactly one prime ideal, it is maximal, since \(A\) has to have at least one maximal ideal.
Moreover, the nilradical is the intersection of all prime ideals, and hence is this ideal.
Since every non unit in \(A\) is contained in a maximal ideal, which is the nil radical, every non unit is nilpotent.

Now suppose every element is either a unit or nilpotent.
Recall that \(A/\nilrad\) has no nilpotent elements.
Let \(x \in A/\nilrad\) be nonzero.
Then \(x \notin A/\nilrad\), and hence is a unit in \(A\).
Thus \(A/\nilrad\) is a field.

Finally suppose \(A/\nilrad\) is a field.
Then \(\nilrad\) is maximal.
However, it is equal to the intersection of all prime ideals.
Thus it is the only prime ideal in \(A\).

\begin{exercise} \label{ex1-11}
    A ring \(A\) is \textit{Boolean} if \(x^2 = x\) for all \(x \in A\).
    In a Boolean ring \(A\), show that
    \begin{enumerate}
        \item
            \(2x = 0\) for all \(x \in A\);
        \item
            every prime ideal \(\idealp\) is maximal, and \(A/\idealp\) is a field of two elements;
        \item
            every finitely generated ideal is principal.
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate}
    \item
        If \(x^2 = x\) then \(x + 1 = (x + 1)^2 = x^2 + 2x + 1 = x + 2x + 1\).
        Hence \(2x = 0\).
    \item
        Every prime is maximal from Exercise \ref{ex1-7}.
        Let \(\idealp\) be such a prime.
        Let \(x + \idealp\) be nonzero.
        Then \(x^2 - x \in \idealp\).
        Hence \(x - 1 \in \idealp\).
        Thus \(x \equiv 1 \pmod{\idealp}\), so \(A/\idealp \cong \FF_2\).
    \item
        Let \(\ideala\) be generated by \(a_1, a_2\).
        Let \(a = a_1 + a_2 - a_1 a_2\).
        Then \(a_1 a = a_1 + a_1 a_2 - a_1 a_2 = a_1\) and similarly \(a_2 a = a_2\).
        Every \(x \in \ideala\) can be written as \(x = x_1 a_1 = x_2 a_2 = x_1 a_1 a + x_2 a_2 a\).
        Thus \(\ideala\) is generated by \(a\).

        If \(\ideala\) is generated by \(a_1, \cdots, a_{n-1}, a_n\), by induction the \(a_1, \cdots, a_{n-1}\) generate a principal ideal, and so we can repeat the above argument.
\end{enumerate}

\begin{exercise} \label{ex1-12}
    A local ring contains no idempotents \(\neq 0,1\).
\end{exercise}

\noindent\textbf{Solution:}
Let \(\idealm\) be the maximal ideal.
Let \(e\) be an idempotent in \(A\).
If \(e \in \idealm\) then \(1 - e\) is a unit, as \(\idealm\) is the Jacobson radical.
If \(c\) is the inverse, then \(1 = c - ce\), and then \(c = 1 - ce\).
Then \(1 = (1 - e)(1 - ce) = 1 - e - ce + ce = 1 - e\), so \(e = 0\).
If \(e \notin \idealm\), then \(e\) is a unit.
Since \(e^2 - e = e (e-1) = 0\), we get that \(e = 1\).

\begin{exercise}
    In a ring \(A\), let \(\Sigma\) be the set of all ideals in which every element is a zero-divisor.
    Show that the set \(\Sigma\) has maximal elements and that every maximal element of \(\Sigma\) is a prime ideal.
    Hence, the set of zero-divisor in \(A\) is a union of prime ideals.
\end{exercise}

\noindent\textbf{Solution:}
We apply Zorn's lemma.
Firstly, the zero ideal is in \(\Sigma\) so it is non-empty.
Let \(C\) be a chain.
Then the union is an ideal.
Every element in the union a zero-divisor, so the union is an upper bound of \(C\).
Hence maximal elements exist.
Let \(\idealp\) be a maximal element.
Assume \(ab \in \idealp\) but \(a, b \notin \idealp\).
Then \((a) + \idealp\) and \((b) + \idealp\) are strictly larger then \(\idealp\) and hence contain non-zero divisors \(aa' + p, bb' + p\).
However \((aa' + p)(bb' + p) = aba'b' + aa'p + bb'p + p^2\) and this is a zero-divisor.
This is a contradiction (if \(abc = 0\) then \(a\) has annihilator \(bc\))!
So \(\idealp\) is prime.

\begin{exercise} \label{ex1-15}
    Let \(A\) be a ring and let \(X\) be the set of all prime ideals of \(A\).
    For each subset \(E\) of \(A\), let \(V(E)\) denote the set of prime ideals of \(A\) which contain \(E\).
    Prove that
    \begin{enumerate}
        \item
            if \(\ideala\) is the ideal generated by \(E\), then \(V(E) = V(\ideala) = V(\rad{\ideala})\).
        \item
            \(V(0) = X\), \(V(1) = \varnothing\).
        \item
            if \((E_i)_{i \in I}\) is any family of subsets of \(A\), then
            \[
                V(\bigcup_{i \in I} E_i) = \bigcap_{i \in I} V(E_i)
            \]
        \item
            \(V(\ideala \cap \idealb) = V(\ideala \idealb) = V(\ideala) \cup V(\idealb)\) for any ideals \(\ideala, \idealb\) of \(A\).
    \end{enumerate}
    These results show that the sets \(V(E)\) satisfy the axioms for closed sets in a topological space.
    The resulting topology is called the \textit{Zariski topology}.
    The topological space \(X\) is called the \textit{prime spectrum} of \(A\), and is written \(\Spec(A)\).
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate}
    \item
        If \(E \subseteq \idealp\) then \(\ideala \subseteq \idealp\) since \(\ideala\) is the smallest ideal containing \(E\).
        Conversely if \(\ideala \subseteq \idealp\) then \(E \subseteq \ideala \subseteq \idealp\).
        Hence \(V(E) = V(\ideala)\).
        Next, if \(\ideala \subseteq \idealp\) then \(\rad{\ideala} \subseteq \rad{\idealp} = \idealp\).
        Conversely if \(\rad{\ideala} \subseteq \idealp\) then \(\ideala \subseteq \rad{\ideala} \subseteq \idealp\).
        Hence \(V(\ideala) = V(\rad{\ideala})\).
    \item
        Every prime ideal contains \(0\) and no prime ideals contain \(1\) as they are proper.
        Hence \(V(0) = X\) and \(V(1) = \varnothing\).
    \item
        \(\bigcup_{i \in I} E_i \subseteq \idealp\) iff \(E_i \subseteq \idealp\) for each \(i \in I\).
        Hence \(V(\bigcup_{i \in I} E_i) = \bigcap_{i \in I} V(E_i)\).
    \item
        If \(\ideala \cap \idealb \subseteq \idealp\) then \(\ideala \idealb \subseteq \ideala \cap \idealb \subseteq \idealp\).
        Conversely, if \(\ideala \idealb \subseteq \idealp\) then either \(\ideala \subseteq \idealp\) or \(\idealb \subseteq \idealp\), and hence \(\ideala \cap \idealb \subseteq \idealp\).
        Hence \(V(\ideala \cap \idealb) = V(\ideala \idealb)\).
        Next, if \(\ideala\idealb \subseteq \idealp\) then (as stated) either \(\ideala \subseteq \idealp\) or \(\idealb \subseteq \idealp\).
        Conversely if \(\ideala \subseteq \idealp\) or \(\idealb \subseteq \idealp\), then \(\ideala \cap \idealb \subseteq \idealp\).
        Hence \(V(\ideala \idealb) \subseteq V(\ideala) \cup V(\idealb)\) and \(V(\ideala) \cup V(\idealb) \subseteq V(\ideala \cap \idealb)\).
\end{enumerate}

\begin{exercise} \label{ex1-17}
    For each \(f \in A\), let \(X_f\) denote the complement of \(V(f)\) in \(X = \Spec(A)\).
    The sets \(X_f\) are open.
    Show that they form a basis of open sets for the Zariski topology, and that
    \begin{enumerate}
        \item
            \(X_f \cap X_g = X_{fg}\);
        \item
            \(X_f = \varnothing\) \(\iff\) \(f\) is nilpotent;
        \item
            \(X_f = X\) \(\iff\) \(f\) is a unit;
        \item
            \(X_f = X_g\) \(\iff\) \(\rad{(f)} = \rad{(g)}\);
        \item
            X is quasi-compact (that is, every open covering of \(X\) has a finite subcovering).
        \item
            More generally, each \(X_f\) is quasi-compact.
        \item
            An open subset of \(X\) is quasi-compact if and only if it is a finite union of sets \(X_f\).
    \end{enumerate}
    The sets \(X_f\) are called \textit{basic open sets} of \(X = \Spec(A)\).
\end{exercise}

\noindent\textbf{Solution:}
Note that \(\idealp \in X_f\) iff \(f \notin \idealp\).
Every \(\idealp \in X\) is proper, and so some \(f \in X - \idealp\), so \(\idealp \in X_f\).
Hence \(X_f\) covers \(X\).
Next, if \(\idealp \in X_f \cap X_g\), then \(\idealp \in X_{fg}\) as we will show next.
Hence the \(X_f\) form a basis!
\begin{enumerate}
    \item
        Note that since \(\idealp\) is prime, \(f, g \notin \idealp\) iff \(fg \notin \idealp\).
        Hence \(X_f \cap X_g = X_{fg}\) directly!
    \item
        \(X_f = \varnothing\) iff \(f \in \idealp\) for every \(\idealp \in \Spec(A)\), iff \(f\) is in the nilradical.
    \item
        If \(X_f = X\) then \(f\) is not contained in any prime ideal.
        Hence it is not contained in any maximal ideal, so it is a unit.
        Conversely, suppose \(f\) is a unit.
        Then \((f) = A\), so \(X_f = X\).
    \item
        If \(X_f = X_g\) then \(f \in \idealp\) iff \(g \in \idealp\).
        Recall that \(\rad{(f)}\) is the intersection of all prime ideals containing \((f)\).
        But a prime contains \((f)\) iff it contains \((g)\), so \(\rad{(f)} = \rad{(g)}\).
        Conversely, if \(\rad{(f)} = \rad{(g)}\), then \(V(f) = V(\rad{(f)}) = V(\rad{(g)}) = V(g)\).
    \item
        Let \(X\) be covered by \((X_{f_i})_{i \in I}\)s.
        Then \(\bigcap_{i \in I} V(f_i) = V(\sum_{i \in I} (f_i)) = \varnothing\) by Exercise \ref{ex1-15}.
        Since every maximal ideal is prime, this necessarily implies that \(\sum_{i \in I} (f_i) = (1)\).
        Hence \(1 = \sum_{j \in J} f_j g_j\) for some finite \(J\).
        Then \(\sum_{j \in J} (f_j) = 1\), so \(X\) is covered by \((X_{f_j})_{j \in J}\).
    \item
        Let \(X_f\) be covered by \((X_{fg_i})_{i \in I}\).
        Then \(V(\sum_{i \in I} (f g_i)) = V(f)\).
        Hence a prime ideal contains \(f\) iff it contains \(\sum_{i \in I} (f g_i)\).
        Hence their radicals are equal.
        Hence \((f) \subseteq \rad{\sum_{i \in I} (f g_i)}\).
        This means that \(f^n = \sum_{j \in J} f g_j h_j\) for some finite set \(j\) and some \(n\).
        If \(\idealp\) is a prime ideal containing \(\sum{j \in J} (f g_j)\), then \(f^n \in \idealp\), so \(f \in \idealp\).
        Hence \(V(\sum_{j \in J} (f g_j)) \subseteq V(f)\).
        Thus \((X_{f g_j})_{j \in J}\) covers \(X_f\).
    \item
        Suppose \(U \subseteq X\) is quasi-compact.
        Take \((X_{f_i})_{i \in I}\) so that \(U = \bigcup_{i \in I} X_{f_i}\) from the basis.
        Then there is a finite subcover \((X_{f_j})_{j \in J}\).
        Hence \(U\) is a finite union of \(X_f\) sets.

        The converse is just a fact in topology, that the finite union of quasi-compact sets is quasi-compact.
        Indeed take a finite set \((X_{f_j})_{j \in J}\).
        Let \(U = \bigcup_{j \in J} X_{f_j}\) and take a cover \((X_i)_{i \in I}\).
        Then the \((X_i)\) cover each \(X_{f_j}\), and hence each have a finite subcover.
        Their union is finite and covers \(U\).
\end{enumerate}

\begin{exercise}

            If \(Y\) is an irreducible (Exercise \ref{ex1-19}) subspace of \(X\), then the closure \(\overline{Y}\) of \(Y\) in \(X\) is irreducible.
        \item

        Hence they interesct, so \(U\) and \(V\) intersect.

        Therefore the union of \(C\) is irreducible, and thus is an upper bound of \(C\).
    \item
        If \(Y\) is a maximal irreducible subspace, then \(\overline{Y}\) is irreducible, and hence equal to \(Y\), so \(Y\) is closed.

\end{enumerate}

\begin{exercise}
    Let \(A = \prod_{i=1}^n A_i\) be the direct product of rings \(A_i\).
    Show that \(\Spec(A)\) is the disjoint union of open (and closed) subspaces \(X_i\), where \(X_i\) is canonically homeomorphic to \(\Spec(A_i)\).
    Conversely, let \(A\) be any ring.
    Show that the following are equivalent:
    \begin{enumerate}
        \item
            \(X = \Spec(A)\) is disconnected.
        \item
            \(A = A_1 \times A_2\) where neither rings \(A_1, A_2\) is the zero ring.
        \item
            \(A\) contains an idempotent \(\neq 0, 1\),
    \end{enumerate}
    In particular, the spectrum of a local ring is always connected (Exercise \ref{ex1-12}).
\end{exercise}

\noindent\textbf{Solution:}
Let \(A = \prod_{i=1}^n A_i\).
Since the product of rings is an integral domain iff one of the rings is the integral domain and the rest are zero, and since products commute with quotients, a prime ideal of \(A\) is precisely a product of \(A_i\)s except for one component, which is a prime ideal.
We will show that \(X = \Spec(A)\) satisfies the universal property of the disjoint union of \(X_i\).
Let \(f_i : X_i \to Y\) be a choice of continuous maps, we aim to show there is a unique \(f : X \to Y\) such that it commutes with the \(f_i\) maps through the maps \(\pi_i^* : X_i \to X\) induced by the quotient map.
Let \(x \in X\), then \(x = A_1 \times \dots \times \idealq \times \dots \times A_n\) for a prime ideal \(\idealq\) of \(A_i\).
Hence \(x = \pi_i^*(q)\), so \(f(x) = f_i(\idealq)\) is the unique choice of mapping that we can make.
It is continuous, as if \(U\) is open in \(Y\), then \(f^{-1}(U) = \bigcup_i f_i^{-1}(U)\) which is open.

Next, suppose \(X = \Spec(A)\) is disconnected.
Let \(V(\ideala), V(\idealb)\) (for proper ideals \(\ideala, \idealb\) of \(A\)) be disjoint clopen subsets of \(X\) so that \(V(\ideala) \cup V(\idealb) = X\).
Then \(V(\ideala) \cap V(\idealb) = \varnothing = V(A)\), so \(\ideala\) and \(\idealb\) are coprime.
Also, \(X = V(0) = V(\ideala) \cup V(\idealb) = V(\ideala \idealb)\), so \(\ideala \idealb \subseteq \nilrad\).
Thus there are \(a \in \ideala, b \in \idealb\) so that \(a + b = 1\) and \(ab \in \nilrad\).
Then \(\rad{(a)} + \rad{(b)} = (1) = \rad{(a^n)} + \rad{(b^n)}\) so \((a^n)\) and \((b^n)\) are coprime.
Hence there is some \(e \in (a^n), e' \in (b^n)\) so that \(e + e' = 1\).
Then \(e e' \in (a^n b^n) = 0\).
Finally, \(e = e (e + e') = e^2\), so \(e\) is an idempotent.
It is not \(1\) since \((a^n)\) is a proper ideal.
It is not zero, since then \(e' = 1\) and \((b^n)\) is a proper ideal.

Finally, suppose \(A\) has an idempontent \(e \neq 0, 1\).
Then \((1 - e)^2 = 1 - 2e + e^2 = 1 - e\), so \(1-e\) is also idempotent.
Then since \(e + (1 - e) = 1\), the ideals \((e)\) and \((1 - e)\) are coprime.
Also, \(e (1-e) = e - e^2 = 0\), so \((e) \cap (1 - e) = (0)\).
Hence by the Chinese remainder theorem, \(A \cong A/(e) \times A/(1-e)\).

\begin{exercise} \label{ex1-23}
    Let \(A\) be a Boolean ring (Exercise \ref{ex1-11}), and let \(X = \Spec(A)\).
    \begin{enumerate}
        \item
            For each \(f \in A\), the set \(X_f\) (Exercise \ref{ex1-17}) is both open and closed in \(X\).
        \item
            Let \(f_1, \dots, f_n \in A\).
            Show that \(X_{f_1} \cup \cdots \cup X_{f_n} = X_f\) for some \(f \in A\).
        \item
            The sets \(X_f\) are the only subsets of \(X\) which are both open and closed.
        \item
            \(X\) is a compact Hausdorff space.
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate}
    \item
        If \(f \notin \idealp\) then \(1 - f + \idealp = 0 + \idealp\) since \(A / \idealp\) is isomorphic to the field of two elements by Exercise \ref{ex1-11}.
        Hence \(1 - f \in \idealp\), so \(\idealp \in V(1 - f)\).
        Conversely if \(\idealp \in V(1-f)\) then \(1 + \idealp = f + \idealp\), so \(f \notin \idealp\) and \(\idealp \in X_f\).
        Hence \(X_f = V(1-f)\).
    \item
        Observe that
        \[
            \bigcup_{i=1}^n X_{f_i} = \bigcup_{i=1}^n V(1 - f_i) = V\left(\bigcap_{i=1}^n (1 - f_i)\right).
        \]
        Then \(\prod_{i=1}^n (1 - f_i) \subseteq \bigcap_{i=1}^n (1 - f_i)\), and this product is finitely generated.
        Hence, it is principal, equal to \((f)\).
        Thus \(\bigcup_{i=1}^n X_{f_i} = V(f)\).
    \item
        Let \(Y\) be clopen in \(X\).
        Then it is quasi-compact by Exercise \ref{ex1-17}.

\begin{exercise} \label{ex1-24}
    Let \(L\) be a lattice, in which the sup and inf of two elements \(a, b\) are denoted by \(a \land b\) and \(a \lor b\) respectively.
    \(L\) is a \textit{Boolean lattice} (or \textit{Boolean algebra}) if
    \begin{enumerate}
        \item
            \(L\) has a least element and a greatest element (denoted \(0, 1\) respectively).
        \item
            Each of \(\lor, \land\) is distributive over the other.
        \item
            Each \(a \in L\) has a unique complement \(a' \in L\) such that \(a \lor a' = 1\) and \(a \land a' = 0\).
    \end{enumerate}
    (For example, the set of all of subsets of a set, ordered by inclusion, is a Boolean lattice.)

    Let \(L\) be a Boolean lattice.
    Define addition and multiplication in \(L\) by the rules
    \[
        a + b = (a \land b') \lor (a' \land b), \quad ab = a \land b.
    \]
    Verify that in this way \(L\) becomes a Boolean ring.

    Conversely, starting with a Boolean ring \(A\), define an ordering on \(A\) as follows: \(a \leq b\) means that \(a = ab\).
    Show that, with respect to this ordering, \(A\) is a Boolean lattice.
    In this way, we obtain a one-to-one correspondence between (isomorphism classes of) Boolean rings and (isomorphism classes) of Boolean lattices.
\end{exercise}

\noindent\textbf{Solution:}
We have to verify a lot of axioms.
Firstly we show de Morgan's law for \(\lor\).
\begin{align*}
    (a \lor b) \lor (a' \land b') = (a \lor b \lor a') \land (a \lor b \lor b') = 1 \land 1 = 1 \\
    (a \lor b) \land (a' \land b') = (a \land a' \land b') \lor (b \land a' \land b') = 0 \lor 0 = 0
\end{align*}
So \((a \lor b)' = a' \land b'\) by uniqueness of complements.
Next we show the de Morgan's law for \(\land\).
\begin{align*}
    (a \land b) \lor (a' \lor b') = (a \lor a' \lor b') \land (b \lor a' \lor b') = 1 \land 1 = 1 \\
    (a \land b) \land (a' \lor b') = (a \land b \land a') \lor (b \land a' \land b') = 0 \lor 0 = 0 
\end{align*}
So \((a \land b)' = a' \lor b'\).
Next we show a useful fact.
\[
    a + b = ab' \lor a'b = (a \lor a'b) (b' \lor a'b) = (a \lor a') (a \lor b) (b' \lor a') (b' \lor b) = (a \lor b) (a' \lor b')
\]
Now \(a + b = (a \lor b) (a' \lor b') = (b \lor a) (b' \lor a') = b + a\) so addition is commutative.
Next,
\begin{align*}
    a (b + c)
    &= a (b \lor c) (b' \lor c') = \\
    &= (ab \lor ac) (b' \lor c') = \\
    &= a'ab \lor a'ac \lor (ab \lor ac) (b' \lor c') \\
    &= a'(ab \lor ac) \lor (ab \lor ac) (b' \lor c') \\
    &= (ab \lor ac) (a' \lor b' \lor c') \\
    &= (ab \lor ac) ((ab)' \lor (ac)') \\
    &= ab + ac
\end{align*}
so multiplication distributes over addition.
Next,
\begin{align*}
    a + (b + c) &= (a (b + c)') \lor (a' (b + c)) \\
                &= (a (b'c' \lor bc)) \lor a' (b \lor c)(b' \lor c') \\
                &= (a b'c' \lor abc) \lor a'(b (b' \lor c') \lor c (b' \lor c')) \\
                &= (a b'c' \lor abc) \lor a'(bc' \lor b'c) \\
                &= a b'c' \lor abc \lor a'bc' \lor a'b'c \\
                &= ab'c' \lor a'bc' \lor a'b'c \lor abc \\
                &= (ab' \lor a'b)c' \lor a'b'c \lor abc \\
                &= (a(a' \lor b') \lor b(a' \lor b'))c' \lor a'b'c \lor abc \\
                &= (a \lor b)(a' \lor b')c' \lor a'b'c \lor abc \\
                &= (a \lor b)(a' \lor b')c' \lor (a'b' \lor ab) c \\
                &= (a + b)c' \lor (a + b)' c = (a + b) + c
\end{align*}
so addition is associative (oh my).
Hence this is a ring.
It is boolean, as \(a^2 = a\).

Conversely, we show \(A\) is a Boolean algebra.
Firstly, \(ab = a \land b\) is known from the definition of \(\leq\).
Next, we suppose \(a \lor b = a + b + ab\).
If \(a \leq b\), then \(a = ab\), so \(a + ab = 0\).
Hence \(a + b + ab = a \lor b = b\), so \(a \lor b\) is the least upper bound.
Next, \(0 \leq a\) since \(0a = 0\), and \(a \leq 1\) since \(1a = a\).
Next, \(a (b \lor c) = ab + ac + abc = ab + ac + abac = ab \lor ac\) and
\[
    a \lor bc = a + bc + abc = a + ac + ac + ab + bc + abc + aab + abc + abac = (a + b + ab) (a + c + ac) = (a \lor b) (a \lor c)
\]
noting \(2x = 0\) by Exercise \ref{ex1-11}, so \(\lor\) and \(\land\) distribute.
Finally, for \(a \in A\), we define \(a' = 1-a\).
Then \(a a' = a^2 - a = 0\) and \(a \lor a' = a + 1 - a + a^2 - a = 1\), so \(a'\) is the complement of \(a\).
If \(ab = 0\) and \(a \lor b = a + b + ab = 1\) then \(b = 1 - a = a'\) so \(a'\) is unique.
Hence \(A\) is a Boolean lattice.

We obtain a bijectino between Boolean lattices and Boolean rings as these constructions are inverse.
Indeed for a fixed Boolean algebra \(L\), \(a \leq b\) iff \(a = ab\), and for a fixed Boolean ring \(A\),
\[
    a b' \lor a' b
    = a b' + a' b + a a' b b'
    = a b' + a' b
    = a b' + 2 a' b' + a' b
    = (a + a') b' + a' (b + b')
    = a' + b'
    = a + b
\]
and so we recover the same structure going through the composite of each construction in both directions.

\begin{exercise}

\end{exercise}

\noindent\textbf{Solution:}
Let \(L\) be a Boolean lattice.
Then \(L\) has Boolean ring structure by Exercise \ref{ex1-24}.

\begin{exercise} \label{ex2.16}
    Let \(A, B\) be rings, let \(M\) be an \(A\)-module, \(P\) be a \(B\)-module and \(N\) an \((A, B)\)-bimodule.
    Then \(M \tensor_A N\) is naturally a \(B\)-module, \(N \tensor_B P\) is naturally an \(A\)-module, and we have
    \[
        (M \tensor_A N) \tensor_B P \cong M \tensor_A (N \tensor_B P)
    \]
\end{exercise}

\noindent\textbf{Solution:}
I shall interpret the word naturally to mean that \(- \tensor_A N\) is a functor from \(\Mod_A\) to \(\Mod_B\).
It is sufficient to only show this direction of tensoring, the other direction should be identical.
For each \(b \in B\), define a bilinear map \((m, n) \mapsto m \tensor (bn)\).
This induces a map from \(M \tensor_A N \to M \tensor_A N\).
This multiplication preserves addition and multiplication of the ring structure (indeed \(m \tensor (b'(b n)) = m \tensor ((b' b)n)\) and \(m \tensor ((b + b')n) = m \tensor (bn + b'n)\)).
Hence this makes \(M \tensor_A N\) a \(B\)-module.
Now suppose \(f : M \to M'\) is an \(A\)-module homomorphism.
We obtain a map \(f \tensor N : M \tensor_A N \to M' \tensor N\) bilinearly mapping \((m, n) \mapsto f(m) \tensor n\).
Indeed \(\id \tensor N = \id\) and \((g \circ f) \tensor N = (g \tensor N) \circ (f \tensor N)\) (expand the definition).

I shall interpret this big isomorphism we have to prove as an \((A, B)\)-bimodule isomorphism.
These tensor products are special cases of a general bimodule tensor product described as follows.
If \(M\) is an \(A\)-module and \(N\) is an \((A, B)\)-bimodule, then \(M \tensor_A N\) is an \((A, B)\)-bimodule.
It has the universal property that maps out of it are naturally isomorphic to \(A\)-bilinear maps that are \(B\)-linear in the right component.
Indeed given such a bilinear map \(f : M \times N \to P\), define \(\hat{f} : M \tensor_A N \to P\) with \(A\)-bilinearity.
A bimodule homomorphism is a map that is both an \(A\) and \(B\)-module homomorphism, and here \(\hat{f}\) is already an \(A\)-module homomorphism.
To see that it is a \(B\)-module homomorphism, observe
\[
    \hat{f}(b (m \tensor n)) = \hat{f}(m \tensor bn) = f(m, bn) = b f(m, n) = b \hat{f}(m \tensor n)
\]
and extend by linearity.
This construction is natural in \(P\) since post composition does not affect anything we have done.
An identical construction can be made for an \((A, B)\)-bimodule \(N\) and \(B\)-module \(P\) where we require \(A\)-linearity in the left component.

Now that we have a universal property, we can prove the big isomorphism using the Yoneda lemma.
To define a map out of \((M \tensor_A N) \tensor_B P\), we have to define a bilinear map out of \((M \tensor_A N) \times P\) that is \(A\)-linear in the left component.
That is, we have to define a map out of \(M \tensor_A N \times P\) that is a bimodule homomorphism with \(P\) fixed, and is \(B\)-linear with \(M \tensor_A N\) fixed.
This is the same thing as a map out of \(M \times N \times P\) that is \(A\)-linear in \(M\), \(B\)-linear in \(P\) and linear in both \(A\) and \(B\) in \(N\).
Reversing our steps, this shows that the universal property for \((M \tensor_A N) \tensor_B P\) and \(M \tensor_A (N \tensor_B P)\) are the same.
Hence, they are isomorphic as \((A, B)\)-bimodules.

\begin{exercise}
    If \(f : A \to B\) is a ring homomorphism and \(M\) is a flat \(A\)-module, then \(M_B = B \tensor_A M\) is a flat \(B\)-module.
\end{exercise}

\noindent\textbf{Solution:}
Let
\[
    0 \to N' \to N \to N'' \to 0
\]
be an exact sequence of \(B\)-modules.
Tensoring with \(M_B\) gives
\[
    0 \to (M \tensor_A B) \tensor_B N' \to (M \tensor_A B) \tensor_B N \to (M \tensor_A B) \tensor_B N'' \to 0
\]
but this is isomorphic to the following sequence by exercise \ref{ex2.16}.
\[
    0 \to M \tensor_A (B \tensor_B N') \to M \tensor_A (B \tensor_B N) \to M \tensor_A (B \tensor_B N'') \to 0
\]
This sequence is exact as a sequence of \(A\)-modules, but since kernels and images of maps are the same when regarded as a sequence of \(B\)-modules, this sequence is also exact as a \(B\)-module sequence.
Hence \(M_B\) is flat.

\begin{exercise}
    Show that \((\ZZ/m\ZZ) \tensor_\ZZ (\ZZ/n\ZZ) = 0\) if \(m, n\) are coprime.
\end{exercise}

\noindent\textbf{Solution:}
Let \(A\) be a \(Z\)-module (or abelian group) and let \(f : (\ZZ/m\ZZ) \times (\ZZ/n\ZZ) \to A\) be a bilinear map.
Then \(f(0, 0) = f(m, 0) = m f(1, 0) = f(1, 0)\) and \(f(0, 0) = f(0, n) = n f(0, 1) = f(0, 1)\).
Hence, the induced map \(\hat{f} : (\ZZ/m\ZZ) \tensor_\ZZ (\ZZ/n\ZZ) \to A\) is the zero map.
Thus every map out of \((\ZZ/m\ZZ) \tensor_\ZZ (\ZZ/n\ZZ)\) is the zero map, and \((\ZZ/m\ZZ) \tensor_\ZZ (\ZZ/n\ZZ) \cong 0\).

\begin{exercise} \label{ex2-2}
    Let \(A\) be a ring, \(\ideala\) an ideal, \(M\) an \(A\)-module.
    Show that \(A/\ideala \tensor M\) is isomorphic to \(M/\ideala M\).
\end{exercise}

\noindent\textbf{Solution:}
We will first show that \(\ideala \tensor M \cong \ideala M\).
Define \(f : \ideala \tensor M \to \ideala M\) by \(a \tensor m \mapsto am\).
Similarly define \(g : \ideala M \to \ideala \tensor M\) by \(am \mapsto a \tensor m\).
Both maps are homomorphisms, and are clearly self inverse.

Next consider the exact sequence
\[
    0 \to \ideala \to A \to A/\ideala \to 0.
\]
Since tensor is right exact, we obtain the following exact sequence
\[
    \ideala M \to M \to A/\ideala \tensor M \to 0.
\]
The left map is given from sending \(a m \mapsto a m\) and so is injective.
Hence we can extend the sequence to the left with a zero.
\[
    0 \to \ideala M \to M \to A/\ideala \tensor M \to 0
\]
Thus \(A/\ideala \tensor M \cong M/\ideala M\).

\begin{exercise}
    Let \(A\) be a local ring, \(M\) and \(N\) finitely generated \(A\)-modules.
    Prove that if \(M \tensor N = 0\) then \(M = 0\) or \(N = 0\).
\end{exercise}

\noindent\textbf{Solution:}
Let \(\idealm\) be the maximal ideal of \(A\) and call its residue field \(k\).
Then \(\idealm\) is the Jacobson radical of \(A\), so the hypotheses of Nakayama's lemma apply.
Exercise \ref{ex2-2} gives us that \(M_k \cong M/\idealm M, N_k \cong N/\idealm N\).
By right exactness of tensor, the map \(M \tensor N \twoheadrightarrow M_k \tensor N\) is surjective, and similarly \(M_k \tensor N \twoheadrightarrow M_k \tensor N_k\) is, so their composite is.
Hence \(M_k \tensor N_k = 0\), so either \(M_k\) or \(N_k\) are the zero vector space.
Hence either \(M = \idealm M\) or \(N = \idealm N\), so by Nakayama's lemma either \(M = 0\) or \(N = 0\).

\begin{exercise} \label{ex2-3}

    Prove that \(M\) is flat \(\iff\) each \(M_i\) is flat.

\noindent\textbf{Solution:}
We first show that a direct sum of short sequences is exact iff each sequence is exact.
Let

    \begin{tikzcd}

    \end{tikzcd}

be exact for each \(i \in I\).
Then \(\ker\left(\bigoplus_i f_i\right) = \bigoplus_i \ker(f_i) = \bigoplus_i \im(f_i) = \im\left(\bigoplus_i f_i\right)\), so the sequence
\[
    \begin{tikzcd}
        \bigoplus_i A_i \arrow[r, "\bigoplus_i f_i"] & \bigoplus_i B_i \arrow[r, "\bigoplus_i g_i"] & \bigoplus_i C_i
    \end{tikzcd}
\]
of direct sums is exact.
Conversely, suppose this direct sum sequence is exact and consider the following commutative diagram.
\[
    \begin{tikzcd}
        A_i \arrow[r, "f_i"] \arrow[d, tail] & B_i \arrow[r, "g_i"] \arrow[d, tail] & C_i \arrow[d, tail] \\
        \bigoplus_i A_i \arrow[r, "\bigoplus_i f_i"] & \bigoplus_i B_i \arrow[r, "\bigoplus_i g_i"] & \bigoplus_i C_i
    \end{tikzcd}
\]
If \(b \in \im(f_i)\), then there is some \(a \in A_i\) so that \(f_i(a) = b\).
Then the image of \(a\) into the direct sum maps to zero in the \(C\) term, so by injectivity \(g_i(b) = 0\), hence \(b \in \ker(f_i)\).
Conversely, if \(b \in \ker(g_i)\), then by commutativity, the image of \(b\) in the direct sum of \(B_i\) is in the kernel of its right arrow.
Hence it is in the image of the direct sum of \(f_i\).
But because of how the direct sum of maps is defined, the term in the direct sum of \(A_i\) must necessarily be zero everywhere but the \(i\)th component, hence in \(A_i\).
So \(b \in \im(f_i)\), so the sequence of \(i\)th modules is exact.

From here we conclude the result of the exercise.
Suppose
\[
    \begin{tikzcd}
        M \tensor N' \arrow[r] & M \tensor N \arrow[r] & M \tensor N''
    \end{tikzcd}
\]
is exact.
Then since tensor distributes over direct sum, the following is exact.
\[
    \begin{tikzcd}
        \bigoplus_i (M_i \tensor N') \arrow[r] & \bigoplus_i (M_i \tensor N) \arrow[r] & \bigoplus_i (M_i \tensor N'')
    \end{tikzcd}
\]
Hence
\[
    \begin{tikzcd}
        M_i \tensor N' \arrow[r] & M_i \tensor N \arrow[r] & M_i \tensor N''
    \end{tikzcd}
\]
is exact.
Conversely, if each of these \(i\) sequences are exact, then the middle direct sum sequence is exact, which is isomorphic to the first sequence.

\begin{exercise} \label{ex2-6}

\noindent\textbf{Solution:}
We can view \(M[x]\) (as an abelian group) as \(\bigoplus_\NN M\).
Then \(M[x]\) has \(A\)-module structure.
To give \(M[x]\) \(A[x]\)-module structure, we need to define a map \(A[x] \to \End(M[x])\), but we have a map \(A \to \End(M[x])\) already from the \(A\)-module structure of \(M[x\), so we just need to choose an endomorphism for \(x\).
The obvious choice is to use the endomorphism right shifting each term in the direct product by one, or equivalently, multiplying by \(x\).

Define a map from \(M[x]\) to \(A[x] \tensor M\) sending \(m x^n\) to \(x^n \tensor m\) and extending by linearity.
Define another map from \(A[x] \tensor M\) to \(M[x]\) sending \(a x^i \tensor m\) to \(a m x^i\) and extending by linearity.
These maps are obviously homomorphisms, and obviously inverses.

\begin{exercise}
    Let \(0 \to M' \to M \to M'' \to 0\) be an exact sequence of \(A\)-modules.
    If \(M'\) and \(M''\) are finitely generated, then so is \(M\).
\end{exercise}

\noindent\textbf{Solution:}
Consider the following diagram.
\[
    \begin{tikzcd}
        0 \arrow[r] & R^m \arrow[r] \arrow[d, two heads] & R^{m+n} \arrow[r] \arrow[d, "?_2"] & R^n \arrow[r] \arrow[dl, "?_1"] \arrow[d, two heads] & 0 \\
        0 \arrow[r] & M' \arrow[r] & M \arrow[r, two heads] & M'' \arrow[r] & 0
    \end{tikzcd}
\]
We aim to construct \(?_1 : R^n \to M\), which would then induce the map \(?_2 : R^{m+n} \to M\) by the universal property of the coproduct.
A map from \(R^n\) to \(M'\) is a choice of \(n\) elements in \(M'\).
Surjectivity then gives us \(n\) elements in \(M\).
Hence, we obtain a (not necessarily unique) map \(?_2 : R^{m+n} \to M\).
The four lemma then tells us that it is surjeective, hence \(M\) is finitely generated.

\begin{exercise}

    Regarding \(N\) as an \(A\)-module by restriction of scalars, form the \(B\)-module \(N_B = B \tensor_A N\)..

\end{exercise}

\noindent\textbf{Solution:}
Define \(p : N_B \to N\) by \(p(b \tensor y) = by\).
Then \(p\) is surjective, so we obtain the following exact sequence.

    \end{tikzcd}
\]
Note also that \(pg = \id\), so this sequence is split exact.
Using the splitting lemma here feels like cheating, so lets prove the result manually instead.

\begin{exercise} \label{ex2-15}
    In the situation of Exercise \ref{ex2-14}, show that every element of \(M\) can be written in the form \(\mu_i(x_i)\) for some \(i \in I\) and some \(x_i \in M_i\).

Then we can take some \(j\) so that \(i_i \leq j\) for all \(i\).

\begin{exercise}
    Let \((A_i)_{i \in I}\) be a family of rings indexed by a directed set \(I\), and for each pair \(i \leq j\) in \(I\) let \(\alpha_{ij} : A_i \to A_j\) be a ring homomorphism, satisfying conditions \ref{ex2-14a} and \ref{ex2-14b} of Exercise \ref{ex2-14}.
    Regarding each \(A_i\) as a \(\ZZ\)-module we can form a direct limit \(A = \varinjlim A_i\).
    Show that \(A\) inherits a ring structure from the \(A_i\) so that the mappings \(A_i \to A\) are ring homomorphisms.
    The ring \(A\) is the \textit{direct limit} of the system \((A_i, \alpha_{ij})\).

    If \(A = 0\) prove that \(A_i = 0\) for some \(i \in I\).
\end{exercise}

\noindent\textbf{Solution:}
If \(I\) is empty then \(A\) is the zero ring.
Hence we assume \(I\) is non-empty.
We can take the identity element of \(A\) to be \(\alpha_i(1)\) for any \(i \in I\).
This choice of element is independent of choice of \(i\), as indeed if \(k \geq i, j\) then \(\alpha_{ik}(1) = \alpha_{jk}(1)\) as the \(\alpha_{ij}\) are ring homomorphisms.
Next, if \(a, b \in A\) then \(a = \alpha_i(a_i)\) and \(b = \alpha_i(b_i)\) for some \(i \in I\) by virtue of \(I\) being directed.
Hence we can define \(ab = \alpha_i(a_i b_i)\).
This is also independent of choice of \(i\) by exactly the same argument as with identities, if \(k \geq i, j\) then \(\alpha_{ik} (a_i b_i) - \alpha_{jk}(a_j b_j) \in D\), so \(\alpha_i(a_i b_i) = \alpha_k(\alpha_{ik}(a_i b_i)) = \alpha_k(\alpha_{ik}(a_j b_j)) = \alpha_j (a_j b_j)\).
Lastly, notice that we have forced the \(\alpha_i : A_i \to A\) to satisfy the conditions of a ring homomorphism (even if we haven't proven \(A\) is a ring yet).

We must now verify \(A\) under this identity and multiplication satisfy the ring axioms!
They are all immediate. \(1 a = \alpha_i(1 a_i) = \alpha_i) = a\) so left identity holds.
\(ab = \alpha_i(a_i b_i) = \alpha_i(b_i a_i) = ba\) so commutativity holds.
\((ab)c = \alpha_i((a_i b_i) c_i) = \alpha_i(a_i (b_i c_i)) = a(bc)\) so associativity holds.
Finally \(a(b + c) = \alpha_i(a_i (b_i + c_i)) = \alpha_i(a_i b_i + a_i c_i) = ab + ac\).
Hence \(A\) is a ring!

Now if \(A = 0\), then by Exercise \ref{ex2-15} \(0 = \mu_{ij}(1)\) so \(M_j = 0\).
