% 1.12
\begin{exercise}
    \begin{enumerate}
        \item \(\ideala \subseteq (\ideala : \idealb)\)
        \item \((\ideala : \idealb) \idealb \subseteq \ideala\)
        \item \(((\ideala : \idealb) : \idealc) = (\ideala : \idealb \idealc) = ((\ideala : \idealc) : \idealb)\)
        \item \((\bigcap_i \ideala_i : \idealb) = \bigcap_i (\ideala_i : \idealb)\)
        \item \((\ideala : \sum_i \idealb_i) = \bigcap_i (\ideala : \idealb_i)\).
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate}
    \item
        For \(a \in \ideala\), \(a \idealb \subseteq \ideala\) since \(\ideala\) is an ideal.
    \item
        Since \(xb \in (\ideala : \idealb) \idealb\) iff \(x \idealb \subseteq \ideala\) and \(b \in \idealb\), \(xb \in \ideala\).
    \item
        Since \(\idealb \idealc = \idealc \idealb\) we only need to show the first equality.
        Observe that \(x \in ((\ideala : \idealb) : \idealc)\) iff \(x \idealc \subseteq (\ideala : \idealb)\), that is for all \(xc \in x \idealc\) with \(c \in \idealc\), \(xc \idealb \subseteq \ideala\).
        This is true iff \(x \idealc \idealb \subseteq \ideala\).
    \item
        \(x \idealb \subseteq \bigcap_i \ideala_i\) iff \(x \idealb \subseteq \ideala_i\) for all \(i\).
    \item
        Note \(x \sum_i \idealb_i = \sum_i x \idealb_i\).
        Then \(\sum_i x \idealb_i \subseteq \ideala\) iff \(x \idealb_i \subseteq \ideala\) for all \(i\).
\end{enumerate}

% 1.13
\begin{exercise}
    \begin{enumerate}
        \item \label{ex-1.13.1} \(\rad{\ideala} \supseteq \ideala\)
        \item \(\rad{\rad{\ideala}} = \rad{\ideala}\)
        \item \(\rad{\mathfrak{ab}} = \rad{\ideala \cap \idealb} = \rad{\ideala} \cap \rad{\idealb}\)
        \item \(\rad{\ideala} = (1) \iff \ideala = (1)\)
        \item \(\rad{\ideala + \idealb} = \rad{\rad{\ideala} + \rad{\idealb}}\)
        \item if \(\idealp\) is prime, \(\rad{\idealp^n} = \idealp\) for all \(n > 0\).
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate}
    \item
        If \(x \in \ideala\), then \(x^1 \in \ideala\) so \(x \in \rad{\ideala}\).
    \item
        \(x \in \rad{\rad{\ideala}}\) iff \((x^n)^m \in \ideala\) iff \(x^{nm} \in \ideala\) iff \(x \in \rad{\ideala}\).
    \item
        If \(x^n \in \ideala \idealb\) then \(x^n \in \ideala \cap \idealb\).
        If \(x^n \in \ideala \cap \idealb\) then \(x^n \in \ideala\) and \(x^n \in \idealb\).
        If \(x^n \in \ideala\) and \(x^m \in \idealb\) then \(x^{mn} \in \ideala \idealb\).
    \item
        If \(1^n \in \ideala\) then \(1 \in \ideala\).
        The converse comes from \ref{ex-1.13.1}.
    \item
        By 1., \(\rad{\ideala + \idealb} \subseteq \rad{\rad{\ideala} + \rad{\idealb}}\).
        If \(x^n = y + z\) where \(y^k \in \ideala\) and \(z^m \in \idealb\) then \(x^{n (k + m)} = (y + z)^{k + m}\) has every term in its binomial expansion with either its \(y\) term at least \(k\) or its \(z\) term at least \(m\), and so \(x^{n (k+m)} \in \ideala + \idealb\).
    \item
        If \(x^m \in \idealp^n\) then \(x^m \in \idealp\) so \(x \in \idealp\).
        Conversely if \(x \in \idealp\) then \(x^n \in \idealp^n\) so \(x \in \rad{\idealp^n}\).
\end{enumerate}

\begin{exercise}
    If \(\ideala_1, \ideala_2\) are ideals of \(A\) and if \(\idealb_1, \idealb_2\) are ideals of \(B\), then
    \[
        \begin{array}{c c}
            (\ideala_1 + \ideala_2)^e = \ideala_1^e + \ideala_2^e, &
            (\idealb_1 + \idealb_2)^c \supseteq \idealb_1^c + \idealb_2^c, \\
            (\ideala_1 \cap \ideala_2)^e \subseteq \ideala_1^e \cap \ideala_2^e, &
            (\idealb_1 \cap \idealb_2)^c = \idealb_1^c \cap \idealb_2^c, \\
            (\ideala_1 \ideala_2)^e = \ideala_1^e \ideala_2^e, &
            (\idealb_1 \idealb_2)^c \supseteq \idealb_1^c \idealb_2^c, \\
            (\ideala_1 : \ideala_2)^e \subseteq (\ideala_1^e : \ideala_2^e), &
            (\idealb_1 : \idealb_2)^c \subseteq (\idealb_1^c : \idealb_2^c), \\
            \rad{\ideala}^e \subseteq \rad{\ideala^e} &
            (\rad{\idealb})^c = \rad{\idealb^c}.
        \end{array}
    \]
\end{exercise}

\noindent\textbf{Solution:}
Note that extension is monotone.
Hence \(\ideala_1^e \subseteq (\ideala_1 + \ideala_2)^e\) and \(\ideala_2^e \subseteq (\ideala_1 + \ideala_2)^e\).
Hence \(\ideala_1^e + \ideala_2^e \subseteq (\ideala_1 + \ideala_2)^e\).
Conversely, let \(\sum_i f(a_i + b_i) \in (\ideala_1 + \ideala_2)^e\).
Then \(\sum_i f(a_i + b_i) = (\sum_i f(a_i)) + (\sum_i f(b_i)) \in \ideala_1^e + \ideala_2^e\).
Next, notice that \((\ideala_1 \cap \ideala_2)^e \subseteq \ideala_1^e\) and \((\ideala_1 \cap \ideala_2)^e \subseteq \ideala_2^e\).
Hence \((\ideala_1 \cap \ideala2)^e \subseteq \ideala_1^e \cap \ideala_2^e\).
Next, let \(\sum_i f(a_i b_i) \in (\ideala_1 \ideala_2)^e\).
Then \(\sum_i f(a_i b_i) = \sum_i f(a_i) f(b_i) \in \ideala_1^e \ideala_2^e\).
Conversely let \(\sum_i (\sum_j f(a_{ij})) (\sum_k f(b_{ik})) \in \ideala_1^e \ideala_2^e\).
Then \(\sum_i (\sum_j f(a_{ij})) (\sum_k f(b_{ik})) = \sum_i \sum_j \sum_k f(a_{ij}) f(b_{ik}) = \sum_i \sum_j \sum_k f(a_{ij}b_{ik}) \in (\ideala_1 \ideala_2)^e\).
Next, let \(\sum_i f(a_i) \in (\ideala_1 : \ideala_2)^e\) such that \(a_i \ideala_2 \subseteq \ideala_1\).
Then \((f(a_i))\ideala_2^e = (a_i)^e\ideala_2^e = (a_i \ideala_2)^e \subseteq a_1^e\) so \(f(a_i) \in (\ideala_1^e : \ideala_2^e)\).
Next let \(\sum_i f(a_i) \in \rad{\idealb}^e\)
Then for some \(n_i\), \(f(a_i)^{n_i} \in \ideala^e\) so \(\sum_i f(a_i) \in \rad{\ideala^e}\).

Let \(a + b \in \idealb_1^c + \idealb_2^c\).
Then \(f(a) \in \idealb_1, f(b) \in \idealb_2\) so \(f(a + b) \in \idealb_1 + \idealb_2\) hence \(a + b \in (\idealb_1 + \idealb_2)^c\).
Next let \(b \in (\idealb_1 \cap \idealb_2)^c\).
Then \(f(b) \in \idealb_1 \cap \idealb_2\).
Hence \(b \in \idealb_1^c \cap \idealb_2^c\).
Conversely if \(b \in \idealb_1^c \cap \idealb_2^c\) then \(f(b) \in \idealb_1\) and \(f(b) \in \idealb_2\) so \(b \in (\idealb_1 \cap \idealb_2)^c\).
Next let \(\sum_i a_i b_i \in \idealb_1^c \idealb_2^c\).
Then \(f(a_i) \in \idealb_1\) and \(f(b_i) \in \idealb_2\) so \(f(\sum_i a_i b_i) \in \idealb_1 \idealb_2\).
Hence \(\sum_i a_i b_i \in (\idealb_1 \idealb_2)^c\).
Next let \(b \in (\idealb_1 : \idealb_2)^c\).
Then \(f(b) \idealb_2 \subseteq \idealb_1\), that is for each \(c \in \idealb_2\), \(f(b) c \in \idealb_1\).
If \(c' \in \idealb_2^c\) so that \(f(c') = c\) then \(f(b c') \in \idealb_1\), so \(bc' \in \idealb_1^c\).
Hence \(b \in (\idealb_1^c : \idealb_2^c)\).
Next, if \(b \in (\rad{\idealb})^c\) so that \(f(b)^n \in \idealb\), then \(f(b^n) \in \idealb\) so \(b^n \in \idealb^c\), so \(b \in \rad{\idealb^c}\).
Conversely, if \(b \in \rad{\idealb^c}\) so that \(b^n \in \idealb^c\), then \(f(b)^n \in \idealb\) so \(f(b) \in \rad{\idealb}\) so \(b \in (\rad{\idealb})^c\).

\begin{exercise} \label{ex1-1}
    Let \(x\) be a nilpotent element of a ring \(A\).
    Show that \(1 + x\) is a unit in \(A\).
    Deduce that the sum of a nilpotent element and a unit is a unit.
\end{exercise}

\noindent\textbf{Solution:}
Suppose \(x^n = 0\).
Notice that
\[
    (1 + x)(\prod_{i=0}^n (-1)^i x^i) = 1 + x - x - x^2 + x^2 + x^3 \cdots \pm x^n = 1 \pm x^n = 1.
\]
If \(u\) is a unit then \(u^{-1} x\) is nilpotent, so \(1 + u^{-1} x\) is a unit, hence \(u + x\) is.

\begin{exercise}
    Let \(A\) be a ring and let \(A[x]\) be the ring of polynomials in an indeterminate \(x\), with coefficients in \(A\).
    Let \(f = a_0 + a_1 x + \cdots + a_n x^n\). Prove that
    \begin{enumerate}
        \item \label{ex1-2-1}
            \(f\) is a unit \(\iff\) \(a_0\) is a unit in \(A\) and \(a_1, \cdots, a_n\) are nilpotent.
        \item
            \(f\) is nilpotent \(\iff\) \(a_0, a_1, \cdots, a_n\) are nilpotent.
        \item
            \(f\) is a zero-divisor \(\iff\) there exists \(a \neq 0\) in \(A\) such that \(a f = 0\).
        \item
            \(f\) is said to be primitive if \((a_0, a_1, \cdots, a_n) = (1)\).
            Prove that if \(f, g \in A[x]\), then \(fg\) is primitive \(\iff\) f and g are primitive.
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate} \label{ex1-2}
    \item
        If \(a_0\) is a unit and \(a_i\) are nilpotent for \(i \geq 1\), then \(a_i x^i\) are nilpotent and their sum is nilpotent, hence by \ref{ex1-1} \(f\) is a unit.

        Conversely suppose \(f\) is a unit with inverse \(g\) (coefficients indexed with \(b_i\)).
        Then \(a_0 b_0 = 1\).
        Next let \(\idealp\) be a prime ideal.
        The map \(\varphi : A[x] \to (A/\idealp)[x]\) has \(\varphi(f)\varphi(g) = 1\).
        Hence the degree of \(\varphi(f)\) and \(\varphi(g)\) are both \(0\) since \((A/\idealp)[x]\) is an integral domain.
        Thus \(\phi(a_i) = 0\) for all \(i \geq 1\), that is \(a_i \in \idealp\).
        Since this is for any prime ideal, the \(a_i\) are in all prime ideals, and hence are nilpotent.
    \item
        Similarly to above, if the \(a_i\) are nilpotent, their product with indeterminates are, and their sum is.

        Let \(f\) be nilpotent with \(f^n = 0\).
        Then \(a_0^n = 0\) so \(a_0\) is nilpotent.
        By exercise \ref{ex1-1}, \(f + 1\) is a unit, and by part \ref{ex1-2-1} the \(a_i\) are nilpotent for \(i \geq 1\).
    \item
        Let \(g \neq 0\) annihilate \(f\) with minimum degree \(m\).
        Then \(b_m a_n = 0\).
        Hence \(a_n g\) has degree less than m, and moreover \(a_n g\) annihilates \(f\).
        Hence \(a_n g = 0\).
        Now suppose for induction that \(a_{n-i} g = 0\) holds for all \(i \leq r\).
        Then \(a_{n-r-1} b_i = - \sum_{j > i} a_{n-j} b_j = 0\).
        Hence for all \(i\), \(a_i b_n = 0\).
        Thus \(b_n f = 0\).
    \item
        Suppose \(fg\) is primitive, and call its coefficients \(c_i\).
        It is sufficient to show \(f\) is primitive.
        Notice that \((a + b) \subseteq (a) + (b)\).
        Hence \((c_i) \subseteq \sum_{j \leq i} (a_j b_{i-j}) \subseteq \sum_{j \leq i} (a_j)\) for all \(i\).
        Thus \((1) = \sum_i (c_i) \subseteq \sum_i (a_j)\), and so \(f\) is primitive.

        Conversely suppose \(f\) and \(g\) are primitive.
        We will show that \(\idealc = (c_0, c_1, \cdots, c_k)\) is not contained in any prime ideal of \(A\).
        Suppose it were contained \(\idealp\).
        Then under the map \(\varphi : A[x] \to (A/\idealp)[x]\), \(\varphi(c_i) = 0\) and so \(\varphi(fg) = 0\).
        Since \((A/\idealp)[x]\) is a domain, either \(\varphi(f) = 0\) or \(\varphi(g) = 0\).
        However, since \((a_0, a_1, \cdots, a_n) \not\supseteq \idealp\) (since \(\idealp\) is proper), some \(a_i\) is not sent to zero.
        Similarly, some \(b_i\) is not sent to zero.
        This is a contradiction, and so \(\idealc\) is not contained in any prime ideal.
        Since every proper ideal is contained in a maximal (and hence prime) ideal, \(\idealc\) is not proper.
\end{enumerate}

\begin{exercise}
    Generalise the results in Exercise \ref{ex1-2} to a polynomial ring \(A[x_1, \cdots, x_n]\) in several indeterminates.
\end{exercise}

\noindent\textbf{Solution:}
The proofs of part 1, 2, and 4 do not depend on being a single variable, so their proofs can be adapted exactly.
We will only show part 3.
Let \(f\) be a zero-divisor with annihilator \(g\).
Let \(d\) be greater than the maximum exponent of indeterminates in \(f\).
Consider the map \(\varphi : A[x_1, \cdots, x_n] \to A[y]\) sending \(x_i\) to \(y^{d^i}\).
Notice that \(\varphi\) restricted to the set of polynomials with maximum exponent indeterminates less than \(d\) is injective.
To see this, write the exponent of the image of each monomial in base \(d\), each monomial is mapped distinctly.
Then \(\varphi(fg) = 0\) so \(\varphi(f)\) is a zero-divisor.
Hence there is an \(a \neq 0\) such that \(\varphi(af) = 0\).
By injectivity of the restriction, \(af = 0\).

\begin{exercise}
    In the ring \(A[x]\), the Jacobson radical is equal to the nilradical.
\end{exercise}

\noindent\textbf{Solution:}
Since the set of maximal ideals is a subset of the set of prime ideals, the nilradical is contained in the Jacobson radical.
Suppose \(f\) were not in the Jacobson radical.
Then there is some \(g \in A[x]\) such that \(1 - fg\) is not a unit.
Hence \(-fg\) is not nilpotent by \ref{ex1-1}.
In particular, \(f\) is not nilpotent, so \(f\) is not in the nilradical.

\begin{exercise}
    Let \(A\) be a ring and let \(A[[x]]\) be the ring of formal power series \(f = \sum_{n=0}^\infty a_n x^n\) with coefficients in \(A\).
    Show that
    \begin{enumerate}
        \item \label{ex1-5-1}
            \(f\) is a unit in \(A[[x]]\) \(\iff\) \(a_0\) is a unit in \(A\).
        \item
            If \(f\) is nilpotent, then \(a_n\) is nilpotent for all \(n \geq 0\).
            Is the converse true?
        \item
            \(f\) belongs to the Jacobson radical of \(A[[x]]\) \(\iff\) \(a_0\) belongs to the Jacobson radical of \(A\).
        \item
            The contraction of a maximal ideal \(\idealm\) of \(A[[x]]\) is a maximal ideal of \(A\), and \(\idealm\) is generated by \(\idealm^c\) and \(x\).
        \item
            Every prime ideal of \(A\) is the contraction of a prime ideal of \(A[[x]]\).
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate}
    \item
        If \(f\) is a unit with inverse \(g = \sum_{n=0}^\infty b_n x^n\) then \(1 = a_0 b_0\) so \(a_0\) is a unit.

        Conversely, suppose \(a_0\) is a unit.
        We define the inverse \(g\) of \(f\) by induction.
        Let \(b_0\) be the inverse of \(a_0\).
        Next, assume we have the first \(n\) terms of \(g\) constructed so that the first \(n\) terms in their product is zero.
        Then \(c_n+1 = \sum_{i=0}^{n+1} a_i b_{n+1-i}\).
        We want \(c_n+1 = 0\) so we get that \(b_{n+1} a_0 = -\sum_{i=1}^{n+1} a_i b_{n+1-i}\).
        Since \(a_0\) is a unit, we can invert it to get \(-b_0 \sum_{i=1}^{n+1} a_i b_{n+1-i}\).
        This completes the construction of the inverse of \(f\).
    \item
        Let \(f\) be such that \(f^n = 0\).
        Then \(a_0^n = 0\) and so \(a_0\) is nilpotent.
        Now suppose for induction that \(a_i\) are nilpotent for all \(i \leq m\) with exponents \(n_i\).
        Since the nilradical is an ideal, we can define \(g = f - \sum_{i \leq m} a_i x^i\) and \(g\) will be nilpotent, say \(g^k = 0\).
        Then \(a_{m+1}^k = 0\), and so \(a_{m+1}\) is nilpotent.

        The converse does not hold, as we could have a power series where each coefficient is nilpotent, but the exponent for it to go to zero is unbounded.
        TODO make an actual example.
    \item
        \(f\) is in the Jacobson radical of \(A[[x]]\) iff \(1 - fg\) is a unit for all \(g \in A[[X]]\).
        This is true iff \(1 - a_0 b_0\) is a unit for all \(b_0 \in A\) by part \ref{ex1-5-1}.
        And finally this is true iff \(a_0\) is in the Jacobson radical of \(A\).
    \item
        Since \(x\) is in the Jacobson radical, it is it \(\idealm\).
        Then since \(A[[x]] / (x) = A\), we see that \(\idealm / (x) = \idealm^c\).
        Hence \(\idealm\) is generated by \(\idealm^c\) and \(x\).
        Moreover \(A[[x]]/\idealm = (A[[x]]/(x))/(\idealm/(x)) = A/\idealm^c\) and so \(\idealm^c\) is maximal.
    \item
        Let \(\idealp\) be a prime ideal of \(A\).
        Then \(\idealp^e + (x)\) is an ideal of \(A[[x]]\) and moreover \(A[[x]]/(\idealp^e + (x)) = A/\idealp\) so \(\idealp^e + (x)\) is prime.
        Moreover \(\idealp\) is the contraction of \(\idealp^e + (x)\).
\end{enumerate}

\begin{exercise}
    A ring \(A\) is such that every ideal not contained in the nilradical contains a nonzero idempotent (that is, an element such that \(e^2 = e \neq 0\)).
    Prove that the nilradical and the Jacobson radical of \(A\) are equal.
\end{exercise}

\noindent\textbf{Solution:}
The nilradical is contained in the Jacobson radical as every maximal ideal is prime, so we just need to show that the opposite inclusion holds.
Suppose \(a\) is in the Jacobson radical but not the nilradical.
Then \((a)\) is an ideal not contained in the nilradical, and so has a nonzero idempotent \(e = ab\) (for some \(b \in A\)).
Note also that since \(a\) is in the Jacobson radical, \(1 - e\) is a unit with inverse \(c\), so \(c - ce = 1\).
Then \(1 - ce = c\), so \(1 = (1 - e)(1 - ce) = 1 - e - ce + ce^2 = 1 - e\).
This contradicts \(e\) being nonzero!
So the Jacobson radical and nilradical are equal.

\begin{exercise} \label{ex1-7}
    Let \(A\) be a ring in which every element \(x\) satisfies \(x^n = x\) for some \(n > 1\) (depending on x).
    Show that every prime ideal is maximal.
\end{exercise}

\noindent\textbf{Solution:}
Let \(\idealp\) be prime.
Let \(x + \idealp\) be a nonzero element.
Then \(x (x^{n-1} - 1) + \idealp = 0 + \idealp\), so \(x^{n-1} \equiv 1 \pmod{\idealp}\).
Hence \(x + \idealp\) is a unit, so \(\idealp\) is maximal.

\begin{exercise}
    Let \(A\) be a ring \(\neq 0\).
    Show that the set of prime ideals of \(A\) has minimal elements with respect to inclusion.
\end{exercise}

\noindent\textbf{Solution:}
We use Zorn's lemma.
Firstly, we know that the set of prime ideals is non-empty since it has a maximal ideal.
Let \(C\) be a chain in the poset of prime ideals.
The intersection \(\idealp = \cap C\) of this chain is an ideal.
Suppose \(a, b \notin \idealp\).
Then there is some \(\idealp_c \in C\) such that \(a, b \notin \idealp_c\).
Hence, \(ab \notin \idealp_c\), so \(ab \notin \idealp\).
Thus \(\idealp\) is prime.
Every chain has a lower bound, so Zorn's lemma tells us that there is a minimal element.

\begin{exercise}
    Let \(\ideala\) be an ideal \(\neq (1)\) in a ring \(A\).
    Show that \(\ideala = \rad{\ideala}\) \(\iff\) \(\ideala\) is an intersection of prime ideals.
\end{exercise}

\noindent\textbf{Solution:}
Notice that \(\rad{\ideala}\) is the contraction of the nilradical in \(A/\ideala\).
The nilradical is the intersection of all prime ideals, and the contraction of a prime ideal is prime.
Hence, \(\rad{\ideala}\) is the intersection of the contraction of all prime ideals in \(A/\ideala\) (which is the prime ideals containing \(\ideala\)).

\begin{exercise}
    Let \(A\) be a ring, \(\nilrad\) its nilradical.
    Show that the following are equivalent:
    \begin{enumerate}
        \item
            \(A\) has exactly one prime ideal;
        \item
            every element of \(A\) is either a unit or nilpotent;
        \item
            \(A/\nilrad\) is a field.
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
If \(A\) has exactly one prime ideal, it is maximal, since \(A\) has to have at least one maximal ideal.
Moreover, the nilradical is the intersection of all prime ideals, and hence is this ideal.
Since every non unit in \(A\) is contained in a maximal ideal, which is the nil radical, every non unit is nilpotent.

Now suppose every element is either a unit or nilpotent.
Recall that \(A/\nilrad\) has no nilpotent elements.
Let \(x \in A/\nilrad\) be nonzero.
Then \(x \notin A/\nilrad\), and hence is a unit in \(A\).
Thus \(A/\nilrad\) is a field.

Finally suppose \(A/\nilrad\) is a field.
Then \(\nilrad\) is maximal.
However, it is equal to the intersection of all prime ideals.
Thus it is the only prime ideal in \(A\).

\begin{exercise}
    A ring \(A\) is \textit{Boolean} if \(x^2 = x\) for all \(x \in A\).
    In a Boolean ring \(A\), show that
    \begin{enumerate}
        \item
            \(2x = 0\) for all \(x \in A\);
        \item
            every prime ideal \(\idealp\) is maximal, and \(A/\idealp\) is a field of two elements;
        \item
            every finitely generated ideal is principal.
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate}
    \item
        If \(x^2 = x\) then \(x + 1 = (x + 1)^2 = x^2 + 2x + 1 = x + 2x + 1\).
        Hence \(2x = 0\).
    \item
        Every prime is maximal from Exercise \ref{ex1-7}.
        Let \(\idealp\) be such a prime.
        Let \(x + \idealp\) be nonzero.
        Then \(x^2 - x \in \idealp\).
        Hence \(x - 1 \in \idealp\).
        Thus \(x \equiv 1 \pmod{\idealp}\), so \(A/\idealp \cong \FF_2\).
    \item
        Let \(\ideala\) be generated by \(a_1, a_2\).
        Let \(a = a_1 + a_2 - a_1 a_2\).
        Then \(a_1 a = a_1 + a_1 a_2 - a_1 a_2 = a_1\) and similarly \(a_2 a = a_2\).
        Every \(x \in \ideala\) can be written as \(x = x_1 a_1 = x_2 a_2 = x_1 a_1 a + x_2 a_2 a\).
        Thus \(\ideala\) is generated by \(a\).

        If \(\ideala\) is generated by \(a_1, \cdots, a_{n-1}, a_n\), by induction the \(a_1, \cdots, a_{n-1}\) generate a principal ideal, and so we can repeat the above argument.
\end{enumerate}

\begin{exercise}
    A local ring contains no idempotents \(\neq 0,1\).
\end{exercise}

\noindent\textbf{Solution:}
Let \(\idealm\) be the maximal ideal.
Let \(e\) be an idempotent in \(A\).
If \(e \in \idealm\) then \(1 - e\) is a unit, as \(\idealm\) is the Jacobson radical.
If \(c\) is the inverse, then \(1 = c - ce\), and then \(c = 1 - ce\).
Then \(1 = (1 - e)(1 - ce) = 1 - e - ce + ce = 1 - e\), so \(e = 0\).
If \(e \notin \idealm\), then \(e\) is a unit.
Since \(e^2 - e = e (e-1) = 0\), we get that \(e = 1\).

\setcounter{exercise}{1}
\begin{exercise}
    \begin{enumerate}
        \item \(\Ann(M + N) = \Ann(M) \cap \Ann(N)\)
        \item \((N : P) = \Ann((N + P)/N)\)
    \end{enumerate}
\end{exercise}

\noindent\textbf{Solution:}
\begin{enumerate}
    \item
        Note that \(a(M + N) = aM + aN\).
        Hence \(a \in \Ann(M + N)\) iff \(aM + aN \subseteq 0\) iff \(aM \subseteq 0\) and \(aN \subseteq 0\) iff \(a \in \Ann(M) \cap \Ann(N)\).
    \item
        If \(aP \subseteq N\) then every \(p \in P\) has \(ap \in N\).
        As \(P \subseteq N + P\) this means \(ap + N = 0 + N\), so \(a \in \Ann((N + P)/N)\).
        Conversely, suppose \(a(n + p) + N = 0 + N\) for all \(n + p \in N + P\).
        Then \(ap + N = 0 + N\), so \(ap \in N\) and so \(a \in (N : P)\).
\end{enumerate}

\setcounter{exercise}{15}
\begin{exercise} \label{ex2.16}
    Let \(A, B\) be rings, let \(M\) be an \(A\)-module, \(P\) be a \(B\)-module and \(N\) an \((A, B)\)-bimodule.
    Then \(M \tensor_A N\) is naturally a \(B\)-module, \(N \tensor_B P\) is naturally an \(A\)-module, and we have
    \[
        (M \tensor_A N) \tensor_B P \cong M \tensor_A (N \tensor_B P)
    \]
\end{exercise}

\noindent\textbf{Solution:}
I shall interpret the word naturally to mean that \(- \tensor_A N\) is a functor from \(\Mod_A\) to \(\Mod_B\).
It is sufficient to only show this direction of tensoring, the other direction should be identical.
For each \(b \in B\), define a bilinear map \((m, n) \mapsto m \tensor (bn)\).
This induces a map from \(M \tensor_A N \to M \tensor_A N\).
This multiplication preserves addition and multiplication of the ring structure (indeed \(m \tensor (b'(b n)) = m \tensor ((b' b)n)\) and \(m \tensor ((b + b')n) = m \tensor (bn + b'n)\)).
Hence this makes \(M \tensor_A N\) a \(B\)-module.
Now suppose \(f : M \to M'\) is an \(A\)-module homomorphism.
We obtain a map \(f \tensor N : M \tensor_A N \to M' \tensor N\) bilinearly mapping \((m, n) \mapsto f(m) \tensor n\).
Indeed \(\id \tensor N = \id\) and \((g \circ f) \tensor N = (g \tensor N) \circ (f \tensor N)\) (expand the definition).

I shall interpret this big isomorphism we have to prove as an \((A, B)\)-bimodule isomorphism.
These tensor products are special cases of a general bimodule tensor product described as follows.
If \(M\) is an \(A\)-module and \(N\) is an \((A, B)\)-bimodule, then \(M \tensor_A N\) is an \((A, B)\)-bimodule.
It has the universal property that maps out of it are naturally isomorphic to \(A\)-bilinear maps that are \(B\)-linear in the right component.
Indeed given such a bilinear map \(f : M \times N \to P\), define \(\hat{f} : M \tensor_A N \to P\) with \(A\)-bilinearity.
A bimodule homomorphism is a map that is both an \(A\) and \(B\)-module homomorphism, and here \(\hat{f}\) is already an \(A\)-module homomorphism.
To see that it is a \(B\)-module homomorphism, observe
\[
    \hat{f}(b (m \tensor n)) = \hat{f}(m \tensor bn) = f(m, bn) = b f(m, n) = b \hat{f}(m \tensor n)
\]
and extend by linearity.
This construction is natural in \(P\) since post composition does not affect anything we have done.
An identical construction can be made for an \((A, B)\)-bimodule \(N\) and \(B\)-module \(P\) where we require \(A\)-linearity in the left component.

Now that we have a universal property, we can prove the big isomorphism using the yoneda lemma.
To define a map out of \((M \tensor_A N) \tensor_B P\), we have to define a bilinear map out of \((M \tensor_A N) \times P\) that is \(A\)-linear in the left component.
That is, we have to define a map out of \(M \tensor_A N \times P\) that is a bimodule homomorphism with \(P\) fixed, and is \(B\)-linear with \(M \tensor_A N\) fixed.
This is the same thing as a map out of \(M \times N \times P\) that is \(A\)-linear in \(M\), \(B\)-linear in \(P\) and linear in both \(A\) and \(B\) in \(N\).
Reversing our steps, this shows that the universal property for \((M \tensor_A N) \tensor_B P\) and \(M \tensor_A (N \tensor_B P)\) are the same.
Hence, they are isomorphic as \((A, B)\)-bimodules.

\setcounter{exercise}{20}
\begin{exercise}
    If \(f : A \to B\) is a ring homomorphism and \(M\) is a flat \(A\)-module, then \(M_B = B \tensor_A M\) is a flat \(B\)-module.
\end{exercise}

\noindent\textbf{Solution:}
Let
\[
    0 \to N' \to N \to N'' \to 0
\]
be an exact sequence of \(B\)-modules.
Tensoring with \(M_B\) gives
\[
    0 \to (M \tensor_A B) \tensor_B N' \to (M \tensor_A B) \tensor_B N \to (M \tensor_A B) \tensor_B N'' \to 0
\]
but this is isomorphic to the following sequence by exercise \ref{ex2.16}.
\[
    0 \to M \tensor_A (B \tensor_B N') \to M \tensor_A (B \tensor_B N) \to M \tensor_A (B \tensor_B N'') \to 0
\]
This sequence is exact as a sequence of \(A\)-modules, but since kernels and images of maps are the same when regarded as a sequence of \(B\)-modules, this sequence is also exact as a \(B\)-module sequence.
Hence \(M_B\) is flat.

\setcounter{exercise}{0}
\renewcommand{\theexercise}{(\arabic{exercise})}

\begin{exercise}
    Show that \((\ZZ/m\ZZ) \tensor_\ZZ (\ZZ/n\ZZ) = 0\) if \(m, n\) are coprime.
\end{exercise}

\noindent\textbf{Solution:}
Let \(A\) be a \(Z\)-module (or abelian group) and let \(f : (\ZZ/m\ZZ) \times (\ZZ/n\ZZ) \to A\) be a bilinear map.
Then \(f(0, 0) = f(m, 0) = m f(1, 0) = f(1, 0)\) and \(f(0, 0) = f(0, n) = n f(0, 1) = f(0, 1)\).
Hence, the induced map \(\hat{f} : (\ZZ/m\ZZ) \tensor_\ZZ (\ZZ/n\ZZ) \to A\) is the zero map.
Thus every map out of \((\ZZ/m\ZZ) \tensor_\ZZ (\ZZ/n\ZZ)\) is the zero map, and \((\ZZ/m\ZZ) \tensor_\ZZ (\ZZ/n\ZZ) \cong 0\).

\begin{exercise}
    Let \(A\) be a ring, \(\ideala\) an ideal, \(M\) an \(A\)-module.
    Show that \(A/\ideala \tensor M\) is isomorphic to \(M/\ideala M\).
\end{exercise}

\noindent\textbf{Solution:}
We will first show that \(\ideala \tensor M \cong \ideala M\).
Define \(f : \ideala \tensor M \to \ideala M\) by \(a \tensor m \mapsto am\).
Similarly define \(g : \ideala M \to \ideala \tensor M\) by \(am \mapsto a \tensor m\).
Both maps are homomorphisms, and are clearly self inverse.

Next consider the exact sequence
\[
    0 \to \ideala \to A \to A/\ideala \to 0.
\]
Since tensor is right exact, we obtain the following exact sequence
\[
    \ideala M \to M \to A/\ideala \tensor M \to 0.
\]
The left map is given from sending \(a m \mapsto a m\) and so is injective.
Hence we can extend the sequence to the left with a zero.
\[
    0 \to \ideala M \to M \to A/\ideala \tensor M \to 0
\]
Thus \(A/\ideala \tensor M \cong M/\ideala M\).
